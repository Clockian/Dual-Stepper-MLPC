Development Enviroment

The IDE the project will use is MPIDE, as it allows for simplest setup.

In MPIDE, open main.pde, which will open all of the other header and class files in the dictionary in separate tabs in the IDE.

The Arduino Board we will use for this project is the Digilent ChipKit Max32 Board along with the ChipKit Basic I/O Shield.

As for specific setting in MPIDE, you will need to change

  • Tools -> Board -> ChipKit Max32
  • Tools -> Serial Port -> COM_ (if it doesn't initially work, change this until it successfully uploads)

Test for Working Board

Use the following code to test that the upload to the board is working correctly, MPIDE is compiling correctly, and such

#include IOShieldOled.h
const int ledPin = 13; // System Operational LED
const int LD1 = 70;
const int LD2 = 71;
const int LD3 = 72;
const int LD4 = 73;
int rotation = 0;

void setup() {
  pinMode(ledPin, OUTPUT);
  pinMode(LD1, OUTPUT);
  pinMode(LD2, OUTPUT);
  pinMode(LD3, OUTPUT);
  pinMode(LD4, OUTPUT);

  IOShieldOled.begin();

  IOShieldOled.clearBuffer();
  IOShieldOled.setCursor(0, 0);
  IOShieldOled.putString("EX00 Basic IO");
  IOShieldOled.setCursor(0, 1);
  IOShieldOled.putString("=============");
  IOShieldOled.setCursor(0, 2);
  IOShieldOled.putString("* Drivers Test *");
  IOShieldOled.setCursor(0, 3);
  IOShieldOled.putString("* 2-19-2014 *");

  IOShieldOled.updateDisplay();
}

void loop() {
  IOShieldOled.updateDisplay();
  digitalWrite(ledPin, HIGH);
  delay(10);

  rotation = rotation + 1;

  if (rotation > 16) { rotation = 1; }

  switch (rotation) {
   case 1: digitalWrite(LD1, HIGH); break;
   case 2: digitalWrite(LD2, HIGH); break;
   case 3: digitalWrite(LD3, HIGH); break;
   case 4: digitalWrite(LD4, HIGH); break;
   case 5: digitalWrite(LD5, HIGH); break;
   case 6: digitalWrite(LD6, HIGH); break;
   case 7: digitalWrite(LD7, HIGH); break;
   case 8: digitalWrite(LD8, HIGH); break;
   case 9: digitalWrite(LD8, LOW); break;
   case 10: digitalWrite(LD7, LOW); break;
   case 11: digitalWrite(LD6, LOW); break;
   case 12: digitalWrite(LD5, LOW); break;
   case 13: digitalWrite(LD4, LOW); break;
   case 14: digitalWrite(LD3, LOW); break;
   case 15: digitalWrite(LD2, LOW); break;
   case 16: digitalWrite(LD1, LOW); break;
  }

  digitalWrite(ledPin, LOW);
  delay(40);
}
